public class EmailController {


    // Email Fields
    public String toEmail { get; set; }
    public String subject { get; set; }
    public String body { get; set; }


    // Attachment Fields
    public Blob fileBody { get; set; }
    public String fileName { get; set; }


    // Method to Send Email
    public PageReference sendEmail() {


        try {


            // Validate Email
            if(!isValidEmail(toEmail)) {


                ApexPages.addMessage(
                    new ApexPages.Message(
                        ApexPages.Severity.ERROR,
                        'Invalid Email ID! Please enter valid email address.'
                    )
                );


                return null;
            }


            // Create Email Message
            Messaging.SingleEmailMessage mail =
                new Messaging.SingleEmailMessage();


            mail.setToAddresses(
                new String[] { toEmail }
            );


            mail.setSubject(subject);
            mail.setPlainTextBody(body);


            // Add Attachment if Selected
            if(fileBody != null) {


                Messaging.EmailFileAttachment efa =
                    new Messaging.EmailFileAttachment();


                efa.setFileName(fileName);
                efa.setBody(fileBody);


                mail.setFileAttachments(
                    new Messaging.EmailFileAttachment[] { efa }
                );
            }


            // Send Email
            Messaging.sendEmail(
                new Messaging.SingleEmailMessage[] { mail }
            );


            // Success Message
            ApexPages.addMessage(
                new ApexPages.Message(
                    ApexPages.Severity.CONFIRM,
                    'Email Sent Successfully!'
                )
            );


            clearForm();


        }
        catch(Exception e) {


            ApexPages.addMessage(
                new ApexPages.Message(
                    ApexPages.Severity.ERROR,
                    'Error : ' + e.getMessage()
                )
            );
        }


        return null;
    }


    // Email Validation Method
    public Boolean isValidEmail(String email) {


        if(String.isBlank(email)) {
            return false;
        }


        String emailRegex =
            '^[A-Za-z0-9+_.-]+@[A-Za-z0-9.-]+$';


        return Pattern.matches(emailRegex, email);
    }


    // Clear Form
    public void clearForm() {


        toEmail = '';
        subject = '';
        body = '';
        fileBody = null;
        fileName = '';
    }
}



<apex:page controller="EmailController">
    <apex:sectionHeader
        title="Email Notification System"
        subtitle="Send Email Using Salesforce" />
    <apex:form enctype="multipart/form-data">
        <apex:pageMessages />
        <apex:pageBlock title="Compose Email">
            <apex:pageBlockSection columns="1">
                <apex:inputText
                    label="Recipient Email"
                    value="{!toEmail}"
                    required="true" />
                <apex:inputText
                    label="Subject"
                    value="{!subject}" />
                <apex:inputTextarea
                    label="Message Body"
                    value="{!body}"
                    rows="6"
                    cols="60" />
                <apex:pageBlockSectionItem>
                    <apex:outputLabel value="Attachment" />
                    <apex:inputFile
                        value="{!fileBody}"
                        filename="{!fileName}" />
                </apex:pageBlockSectionItem>
            </apex:pageBlockSection>
            <apex:pageBlockButtons>
                <apex:commandButton
                    value="Send Email"
                    action="{!sendEmail}" />
            </apex:pageBlockButtons>
        </apex:pageBlock>
    </apex:form>
</apex:page>
